262 /**
263  * @brief Prepares the header of an IPC command structure.
264  * @param cmd IPC command structure.
265  * @param sizeof_raw Size in bytes of the raw data structure to embed inside the IPC request
266  * @return Pointer to the raw embedded data structure in the request, ready to be filled out.
267  */
268 static inline void* ipcPrepareHeader(IpcCommand* cmd, size_t sizeof_raw) {
269  u32* buf = (u32*)armGetTls();
270  size_t i;
271  *buf++ = IpcCommandType_Request | (cmd->NumStaticIn << 16) | (cmd->NumSend << 20) | (cmd->NumRecv << 24) | (cmd->NumExch << 28);
272 
273  u32* fill_in_size_later = buf;
274 
275  if (cmd->NumStaticOut > 0) {
276  *buf = (cmd->NumStaticOut + 2) << 10;
277  }
278  else {
279  *buf = 0;
280  }
281 
282  if (cmd->SendPid || cmd->NumHandlesCopy > 0 || cmd->NumHandlesMove > 0) {
283  *buf++ |= 0x80000000;
284  *buf++ = (!!cmd->SendPid) | (cmd->NumHandlesCopy << 1) | (cmd->NumHandlesMove << 5);
285 
286  if (cmd->SendPid)
287  buf += 2;
288 
289  for (i=0; i<(cmd->NumHandlesCopy + cmd->NumHandlesMove); i++)
290  *buf++ = cmd->Handles[i];
291  }
292  else {
293  buf++;
294  }
295 
296  for (i=0; i<cmd->NumStaticIn; i++, buf+=2) {
297  IpcStaticSendDescriptor* desc = (IpcStaticSendDescriptor*) buf;
298 
299  uintptr_t ptr = (uintptr_t) cmd->Statics[i];
300  desc->Addr = ptr;
301  desc->Packed = cmd->StaticIndices[i] | (cmd->StaticSizes[i] << 16) |
302  (((ptr >> 32) & 15) << 12) | (((ptr >> 36) & 15) << 6);
303  }
304 
305  for (i=0; i<(cmd->NumSend + cmd->NumRecv + cmd->NumExch); i++, buf+=3) {
306  IpcBufferDescriptor* desc = (IpcBufferDescriptor*) buf;
307  desc->Size = cmd->BufferSizes[i];
308 
309  uintptr_t ptr = (uintptr_t) cmd->Buffers[i];
310  desc->Addr = ptr;
311  desc->Packed = cmd->BufferTypes[i] |
312  (((ptr >> 32) & 15) << 28) | ((ptr >> 36) << 2);
313  }
314 
315  u32 padding = ((16 - (((uintptr_t) buf) & 15)) & 15) / 4;
316  u32* raw = (u32*) (buf + padding);
317 
318  size_t raw_size = (sizeof_raw/4) + 4;
319  buf += raw_size;
320 
321  u16* buf_u16 = (u16*) buf;
322 
323  for (i=0; i<cmd->NumStaticOut; i++) {
324  size_t off = cmd->NumStaticIn + i;
325  size_t sz = (uintptr_t) cmd->StaticSizes[off];
326 
327  buf_u16[i] = (sz > 0xFFFF) ? 0 : sz;
328  }
329 
330  size_t u16s_size = ((2*cmd->NumStaticOut) + 3)/4;
331  buf += u16s_size;
332  raw_size += u16s_size;
333 
334  *fill_in_size_later |= raw_size;
335 
336  for (i=0; i<cmd->NumStaticOut; i++, buf+=2) {
337  IpcStaticRecvDescriptor* desc = (IpcStaticRecvDescriptor*) buf;
338  size_t off = cmd->NumStaticIn + i;
339 
340  uintptr_t ptr = (uintptr_t) cmd->Statics[off];
341  desc->Addr = ptr;
342  desc->Packed = (ptr >> 32) | (cmd->StaticSizes[off] << 16);
343  }
344 
345  return (void*) raw;
346 }

402 /**
403  * @brief Parse an IPC command response into an IPC parsed command structure.
404  * @param IPC parsed command structure to fill in.
405  * @return Result code.
406  */
407 static inline Result ipcParse(IpcParsedCommand* r) {
408  u32* buf = (u32*)armGetTls();
409  u32 ctrl0 = *buf++;
410  u32 ctrl1 = *buf++;
411  size_t i;
412 
413  r->IsDomainRequest = false;
414  r->IsDomainResponse = false;
415 
416  r->CommandType = (IpcCommandType) (ctrl0 & 0xffff);
417  r->HasPid = false;
418  r->RawSize = (ctrl1 & 0x1ff) * 4;
419  r->NumHandles = 0;
420 
421  r->NumStaticsOut = (ctrl1 >> 10) & 15;
422  if (r->NumStaticsOut >> 1) r->NumStaticsOut--; // Value 2 -> Single descriptor
423  if (r->NumStaticsOut >> 1) r->NumStaticsOut--; // Value 3+ -> (Value - 2) descriptors
424 
425  if (ctrl1 & 0x80000000) {
426  u32 ctrl2 = *buf++;
427 
428  if (ctrl2 & 1) {
429  r->HasPid = true;
430  r->Pid = *buf++;
431  r->Pid |= ((u64)(*buf++)) << 32;
432  }
433 
434  size_t num_handles_copy = ((ctrl2 >> 1) & 15);
435  size_t num_handles_move = ((ctrl2 >> 5) & 15);
436 
437  size_t num_handles = num_handles_copy + num_handles_move;
438  u32* buf_after_handles = buf + num_handles;
439 
440  if (num_handles > IPC_MAX_OBJECTS)
441  num_handles = IPC_MAX_OBJECTS;
442 
443  for (i=0; i<num_handles; i++)
444  {
445  r->Handles[i] = *(buf+i);
446  r->WasHandleCopied[i] = (i < num_handles_copy);
447  }
448 
449  r->NumHandles = num_handles;
450  buf = buf_after_handles;
451  }
452 
453  size_t num_statics = (ctrl0 >> 16) & 15;
454  u32* buf_after_statics = buf + num_statics*2;
455 
456  if (num_statics > IPC_MAX_BUFFERS)
457  num_statics = IPC_MAX_BUFFERS;
458 
459  for (i=0; i<num_statics; i++, buf+=2) {
460  IpcStaticSendDescriptor* desc = (IpcStaticSendDescriptor*) buf;
461  u64 packed = (u64) desc->Packed;
462 
463  r->Statics[i] = (void*) (desc->Addr | (((packed >> 12) & 15) << 32) | (((packed >> 6) & 15) << 36));
464  r->StaticSizes[i] = packed >> 16;
465  r->StaticIndices[i] = packed & 63;
466  }
467 
468  r->NumStatics = num_statics;
469  buf = buf_after_statics;
470 
471  size_t num_bufs_send = (ctrl0 >> 20) & 15;
472  size_t num_bufs_recv = (ctrl0 >> 24) & 15;
473  size_t num_bufs_exch = (ctrl0 >> 28) & 15;
474 
475  size_t num_bufs = num_bufs_send + num_bufs_recv + num_bufs_exch;
476  r->Raw = (void*)(((uintptr_t)(buf + num_bufs*3) + 15) &~ 15);
477  r->RawWithoutPadding = (void*)((uintptr_t)(buf + num_bufs*3));
478 
479  if (num_bufs > IPC_MAX_BUFFERS)
480  num_bufs = IPC_MAX_BUFFERS;
481 
482  for (i=0; i<num_bufs; i++, buf+=3) {
483  IpcBufferDescriptor* desc = (IpcBufferDescriptor*) buf;
484  u64 packed = (u64) desc->Packed;
485 
486  r->Buffers[i] = (void*) (desc->Addr | ((packed >> 28) << 32) | (((packed >> 2) & 15) << 36));
487  r->BufferSizes[i] = desc->Size;
488  r->BufferTypes[i] = (BufferType) (packed & 3);
489 
490  if (i < num_bufs_send)
491  r->BufferDirections[i] = BufferDirection_Send;
492  else if (i < (num_bufs_send + num_bufs_recv))
493  r->BufferDirections[i] = BufferDirection_Recv;
494  else
495  r->BufferDirections[i] = BufferDirection_Exch;
496  }
497 
498  r->NumBuffers = num_bufs;
499  return 0;
500 }
