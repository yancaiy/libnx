131 /**
132  * @brief Converts a regular service to a domain.
133  * @param[in] s Service object.
134  * @return Result code.
135  */
136 static inline Result serviceConvertToDomain(Service* s) {
137  Result rc = 0;
138  if (serviceIsOverride(s)) {
139  rc = ipcCloneSession(s->handle, 1, &s->handle);
140  if (R_FAILED(rc)) {
141  return rc;
142  }
143  s->type = ServiceType_Normal;
144  }
145  rc = ipcConvertSessionToDomain(s->handle, &s->object_id);
146  if (R_SUCCEEDED(rc)) {
147  s->type = ServiceType_Domain;
148  }
149  return rc;
150 }

152 /**
153  * @brief Closes a service.
154  * @param[in] s Service object.
155  */
156 static inline void serviceClose(Service* s) {
157  switch (s->type) {
158 
159  case ServiceType_Normal:
160  case ServiceType_Domain:
161  ipcCloseSession(s->handle);
162  svcCloseHandle(s->handle);
163  break;
164 
165  case ServiceType_DomainSubservice:
166  serviceCloseObjectById(s, s->object_id);
167  break;
168 
169  case ServiceType_Override:
170  // Don't close because we don't own the overridden handle.
171  break;
172 
173  case ServiceType_Uninitialized:
174  break;
175  }
176 
177  s->type = ServiceType_Uninitialized;
178 }
